macro(ADD_MODULE MODULE)
	add_executable(test_${MODULE} test_${MODULE}.c)
	if (STLIB)
		target_link_libraries(test_${MODULE} relic_s)
	endif(STLIB)
	if (STBIN)
		target_link_libraries(test_${MODULE} "${LFLAGS}")
		set_target_properties(test_${MODULE} PROPERTIES LINK_SEARCH_END_STATIC 1)
	endif(STBIN)
	add_test(test_${MODULE} ${SIMULATOR_PATH} ${SIMULATOR_PARAMS} ${EXECUTABLE_OUTPUT_PATH}/test_${MODULE})
endmacro(ADD_MODULE)
